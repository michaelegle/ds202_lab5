  1. Can you find seasonal trends in accident occurrences in different states? Is summer or winter more dangerous? Why?
states <- map_data('state')

winter <- accident %>%
  filter(MONTH == 12 | MONTH < 3) %>%
  group_by(STATE) %>%
  summarize(n = n())
winter$winter = TRUE

summer <- accident %>%
  filter(MONTH > 6 & MONTH < 9) %>%
  group_by(STATE) %>%
  summarize(n = n())
summer$winter = FALSE

full_join(states, full_join(summer, winter), by = c('group' = 'STATE')) %>%
  ggplot(aes(x = long, y = lat)) +
  geom_polygon(aes(group = group, fill = n)) +
  facet_grid(~ winter) +
  ggtitle('Accidents in the United States (Summer and Winter)')
